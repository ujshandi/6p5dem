jQuery Table AddRow plugin

Description

5tweetsretweet

If you have ever designed a form allowing user entering more than one email? This plugin should help you. It provides the function for adding the add row and delete row dynamicly inside a table:

Update

2010-3-16: Being reported by Simon Wallmann, this plugin cannot work properly in 1.4.2. I have found similar cases in http://forum.jquery.com/topic/jquery-1-4-2-data-unexpected-behaviour and a ticket have been submitted to http://dev.jquery.com/ticket/6166 already. This problem will be fixed in 1.4.3 version. By the way, I've made the adaption to jQuery 1.4.2 before 1.4.3 is launched. If you are going to use jQuery 1.4.2, please make sure to use the latest version 1.02 of this plugin.

2010-3-5: Demonstration on using jQuery Table AddRow Plugin with Rating plugin.

2010-3-5: Demonstration on using jQuery Table AddRow Plugin with Autocomplete plugin

This plugin has been largely updated (adding a lot of parameters, and in the same time, keeping the support of old functionality) on February 19,2010 (Version 1.1). Since I am still busy with the daily job, I will summarize all the parameters later on. For those who want to use the old version (a lighter) please go to the download session for Version 1.0.

This plugin has been updated on October 23,2009 (Version 1.0), with the following enhancement:

  1. For those platform that cannot handle multiple fields with single field name, I have added "inputBoxAutoNumber" parameter.
    • By setting "inputBoxAutoNumber" to true (i.e. $(".addRow").btnAddRow({inputBoxAutoNumber:true});),
    • all the <input> and <textarea> tags' name attribute will be changed automatically.
    • A row number (starting from 1) will be attached after the original name.
    • For example, <input type="text" name="data"/> will be automatically changed to <input type="text" name="data1"/> for the first row, <input type="text" name="data2"/> for the second row and so on.
  2. A new parameter "displayRowCountTo" have been added to allow any <input> with the corresponding classname store the number of row value in it's value attribute. For instance,
    • $(".addRow").btnAddRow({displayRowCountTo:"rowCount"});
    • This instruction told the plugin to calculate the total number of row and store it in any input box with class name "rowCount"
  3. Fix the bug in case similar to Example 1: the new "add" button's value will no longer be cleared.

This plugin has been updated on October 2, 2009, for the following modifications:

  1. Add oddRowCSS and evenRowCSS to define the style in alternative rows inside the table. see #10
  2. Remove the Delete button in last row when using autoAddRow function.

This plugin has been updated on September 29, 2009, for the following modifications:

  1. Add rowNumColumn property, allowing you to define any column to show the row number. See #8
  2. Add autoAddRow(), allowing any table expand automatically without pressing the add button. See #9
  3. Clear textarea in the new row automatic.

This plugin has been updated on September 15, 2009, for the following modifications:

  1. The Delete button will be display if there are more than one rows initially. Thanks for Jeff White for telling me this bug.
  2. The Add Row button will be disabled automatically when the number of row reaches upper limit
  3. Add ignoreClass parameters, see Example #5
  4. Add call back function for further processing see Example #6
  5. ** Add compatibility to other components (defaultText plugin, etc.) with data("init") function, see #7

007.    ExpandableTable.prototype.init=function(target,maxRow){

009.        this.target=$(target).data(className,this);
010.        this.maxRow=maxRow;
011.        this.seed=Math.round(Math.random()*10000);
012.        this.onAddRow=[];
013.        return this
014.    };
015.    ExpandableTable.prototype.live=function(){

017.            var t=this;
018.            this.update();

022.            });
023.            $(".delRow"+this.seed)
024.            .live("click",function(){
025.                var oj=$(this).closest("."+t.cloneClass),
026.                o=oj.clone();
027.                oj.hide().find("*").each(function(i,v){
028.                    if($(v).data("destroy"));
029.                    for(var k in $(v).data("destroy")){
030.                        $(v).data("destroy")[k](v);
031.                    }
032.                });
033.                oj.remove();
034.                $(".addRow"+t.seed)
035.                .attr("disabled",false);
036.                t.update();
037.                if(t.deleteCallBack && $.isFunction(t.deleteCallBack))
038.                    t.deleteCallBack(o);
039.            });
040.            $(".autoAdd"+this.seed)
041.            .live("keyup",function(){
042.                if((this.nodeName.toLowerCase()=="textarea" && $(this).html()!="") ||
043.                (this.nodeName.toLowerCase()=="textarea" && $(this).val()!="") ||
044.                (this.nodeName.toLowerCase()=="input" && $(this).val()!="")) t.addRow();
045.            });
046.            this.goLive=true;
047.        }
048.        return this
049.    };
050.    ExpandableTable.prototype.updateRowNumber=function(){
051.        var t=this;
052.        if(t.rowNumColumn){
053.            $("."+t.cloneClass,t.target).each(function(j,u){
054.                var n=j+1;
055.                $("."+t.rowNumColumn,$(u)).each(function(i,v){
056.                    if($(v).is(":text, textarea"))

060.                });
061.            });
062.        }
063.        return t
064.    };

066.        $("."+this.cloneClass,this.target).each(function(j,t){
067.            var n=j+1;
068.            $("input,textarea",$(t)).each(function(i,v){
069.                if($(v).attr("name")!=""){
070.                    var newName=$(v).attr("name").replace(/\d+$/,"")+n;
071.                    $(v).attr("name",newName);
072.                }
073.            });
074.        });
075.        return this
076.    };

080.            var n=j+1;
081.            $("input,textarea",$(u)).each(function(i,v){
082.                if($(v).attr("id")!=""){

094.        }

103.    };
104.    ExpandableTable.prototype.updateRowCount=function(){
105.        if(this.displayRowCountTo){
106.            var count=$("."+this.cloneClass,this.target).size();
107.            $("."+this.displayRowCountTo,this.target).each(function(i,v){
108.                var nn=v.nodeName.toLowerCase();
109.                if(nn=="input" || nn=="textarea")
110.                    $(v).val(count);
111.                else
112.                    $(v).html(count);
113.            });
114.        }
115.        return this
116.    };
117.    ExpandableTable.prototype.update=function(){
118.        var t=this;
119.        this.delRowButtons=$(".delRow"+this.seed,this.target);
120.        if(this.delRowButtons.size()==1)
121.            this.delRowButtons.hide();
122.        else {
123.            if(this.autoAddRow)
124.                this.delRowButtons.not(":last").show();
125.            else
126.                this.delRowButtons.show();
127.        }
128.        if(this.autoAddRow) {
129.            this.target.find(".autoAdd"+this.seed).removeClass("autoAdd"+t.seed);
130.            this.target
131.            .find("."+t.cloneClass+":last")
132.            .find("input,textarea")
133.            .addClass("autoAdd"+this.seed);
134.        }
135.        if(this.inputBoxAutoNumber) {
136.            this.updateInputBoxName();
137.            this.updateInputBoxId();
138.        }
139.        if(this.inputBoxAutoId) {
140.            this.updateInputBoxId();
141.        }
142.        if(this.hideFirstOnly && this.hideFirstOnly!=""){
143.            $("."+this.cloneClass).eq(0).find("."+this.hideFirstOnly).hide();
144.            $("."+this.cloneClass).not(":first").find("."+this.hideFirstOnly).show();
145.        }
146.        if(this.showFirstOnly && this.showFirstOnly!=""){
147.            $("."+this.cloneClass).eq(0).find("."+this.showFirstOnly).show();
148.            $("."+this.cloneClass).not(":first").find("."+this.showFirstOnly).hide();
149.        }
150.        this.updateRowNumber()
151.        .updateOddRowCSS()

154.        return this
155.    };
156.    ExpandableTable.prototype.addRow=function(){
157.        var newRow;
158.        if(!this.maxRow || (this.maxRow && $("."+this.cloneClass).size()<this.maxRow)){
159.            this.delRowButtons.show();
160.            var lastRow=$("."+this.cloneClass+":last",this.target);
161.            this.newRow=newRow=lastRow.clone();
162.            newRow.find("input:text").val("");
163.            newRow.find("textarea").text("");
164.            if(this.autoAddRow) newRow.find("."+this.cloneClass).hide();
165.            newRow.insertAfter(lastRow);
166.            if(this.ignoreClass && this.ignoreClass!=""){
167.                newRow.find("."+this.ignoreClass).each(function(){
168.                    if(this.nodeName.toLowerCase()=="input" &&
169.                        ($(this).attr("type").toLowerCase()=="text"
170.                        || $(this).attr("type").toLowerCase()=="hidden"
171.                    )) $(this).val("");
172.                    else if(this.nodeName.toLowerCase()=="td") $(this).html(" ");
173.                    else if($(this).html()!="") $(this).text("");
174.                });
175.            }
176.            newRow.find("input:hidden").not("."+this.cloneClass).val("");
177.            if(this.hideFirstOnly && this.hideFirstOnly!=""){
178.                newRow.find("."+this.hideFirstOnly).show();
179.            }
180.            if(this.showFirstOnly && this.showFirstOnly!=""){
181.                newRow.find("."+this.hideFirstOnly).hide();
182.            }
183.            if(this.maxRow && $("."+this.cloneClass).size()>=this.maxRow)
184.                $(".addRow"+this.seed).attr("disabled",true);
185.            this.target.find("."+this.cloneClass+":first").closest("tr").find("*")
186.            .each(function(i,v){
187.                if($(this).data("init")) {
188.                    var jObj=newRow.find("*").eq(i),obj=jObj[0];
189.                    jObj.data("init",{});
190.                    for(var k in $(this).data("init")){
191.                        jObj.data("init")[k]=$(this).data("init")[k];
192.                        jObj.data("init")[k](obj);
193.                    }
194.                }
195.            });
196.            this.update();
